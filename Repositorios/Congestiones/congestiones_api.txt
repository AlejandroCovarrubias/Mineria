CREATE SCHEMA `congestiones_api` DEFAULT CHARACTER SET utf8 ;

CREATE TABLE `congestiones_api`.`congestiones` 
(
  `idcongestion` INT NOT NULL AUTO_INCREMENT,
  
`x` DOUBLE NULL,
  
`y` DOUBLE NULL,

'descripcion' VARCHAR(200) NULL,  
`fecha` DATETIME NULL,
  
PRIMARY KEY (`idcongestion`))

ENGINE = InnoDB

DEFAULT CHARACTER SET = utf8;
